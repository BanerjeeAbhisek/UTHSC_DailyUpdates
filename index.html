The data: metabolomics

Metabolomics measures the thousands of small molecules called metabolites which the body produces through metabolism, giving a chemical snapshot of a person’s state. Each sample (e.g. blood or urine) is profiled for thousands of metabolites, and stacking many samples gives one very large matrix with samples in rows, metabolites in columns.

d3 = require("d3@7")

makeRng = function (seed) {
  let s = seed >>> 0;
  return function () {
    s = (s + 0x6d2b79f5) >>> 0;
    let t = s;
    t = Math.imul(t ^ (t >>> 15), 1 | t);
    t ^= t + Math.imul(t ^ (t >>> 7), 61 | t);
    return ((t ^ (t >>> 14)) >>> 0) / 4294967296;
  };
}

randn = function (rng) {
  let u = 0, v = 0;
  while (u === 0) u = rng();
  while (v === 0) v = rng();
  return Math.sqrt(-2 * Math.log(u)) * Math.cos(2 * Math.PI * v);
}

// Two groups. Big spread along x (drives total variance) but the groups
// only differ along y (the thing we actually care about).
data = {
  const rng = makeRng(42);
  const pts = [];
  const N = 110;
  for (let i = 0; i < N; i++) pts.push({ x: randn(rng) * 2.6, y:  1.3 + randn(rng) * 0.55, g: 0 });
  for (let i = 0; i < N; i++) pts.push({ x: randn(rng) * 2.6, y: -1.3 + randn(rng) * 0.55, g: 1 });
  return pts;
}

meanVec = ({ x: d3.mean(data, d => d.x), y: d3.mean(data, d => d.y) })

// PCA direction = top eigenvector of the 2x2 covariance matrix.
pcaDir = {
  const mx = meanVec.x, my = meanVec.y;
  let a = 0, b = 0, c = 0;
  for (const d of data) { const dx = d.x - mx, dy = d.y - my; a += dx * dx; b += dx * dy; c += dy * dy; }
  const n = data.length; a /= n; b /= n; c /= n;
  const tr = a + c, det = a * c - b * b;
  const lam = tr / 2 + Math.sqrt(Math.max(0, (tr * tr) / 4 - det));
  const v1x = b, v1y = lam - a;
  const v2x = lam - c, v2y = b;
  let vx, vy;
  if (Math.hypot(v1x, v1y) >= Math.hypot(v2x, v2y)) { vx = v1x; vy = v1y; } else { vx = v2x; vy = v2y; }
  const nrm = Math.hypot(vx, vy) || 1;
  return { x: vx / nrm, y: vy / nrm };
}

// PLS direction (single outcome) = normalized covariance of features with the label.
plsDir = {
  const mx = meanVec.x, my = meanVec.y;
  let wx = 0, wy = 0;
  for (const d of data) { const yy = d.g === 0 ? 1 : -1; wx += (d.x - mx) * yy; wy += (d.y - my) * yy; }
  const nrm = Math.hypot(wx, wy) || 1;
  return { x: wx / nrm, y: wy / nrm };
}

kde = function (values, bw, xs) {
  return xs.map(x => {
    let s = 0;
    for (const v of values) { const u = (x - v) / bw; s += Math.exp(-0.5 * u * u); }
    return [x, s / (values.length * bw * Math.sqrt(2 * Math.PI))];
  });
}

// standardized separation between the two groups along a projection direction
separation = function (ts) {
  const t0 = ts.filter((_, i) => data[i].g === 0);
  const t1 = ts.filter((_, i) => data[i].g === 1);
  const m0 = d3.mean(t0), m1 = d3.mean(t1);
  const v0 = d3.variance(t0), v1 = d3.variance(t1);
  const sd = Math.sqrt((v0 + v1) / 2) || 1e-9;
  return Math.abs(m0 - m1) / sd;
}

The challenge

A table with thousands of columns is very difficult to read directly, we just can’t eyeball patterns across that many measurements.

We want to squeeze thousands of numbers into a few meaningful directions that still capture what’s going on. We will perform this using matrix decomposition.


decomp = {
  const n = 24, p = 56, R = 6;
  const rng = (() => { let s = 11 >>> 0; return () => { s = (s + 0x6d2b79f5) >>> 0; let t = s; t = Math.imul(t ^ (t >>> 15), 1 | t); t ^= t + Math.imul(t ^ (t >>> 7), 61 | t); return ((t ^ (t >>> 14)) >>> 0) / 4294967296; }; })();
  const s = [9, 6, 4.5, 3, 2, 1.2];
  const U = [], V = [];
  for (let r = 0; r < R; r++) {
    const ph = rng() * Math.PI * 2, ph2 = rng() * Math.PI * 2;
    const ur = [], vr = [];
    for (let i = 0; i < n; i++) ur.push(Math.cos(Math.PI * (r + 1) * i / n + ph));
    for (let j = 0; j < p; j++) vr.push(Math.cos(Math.PI * (r + 1) * j / (p / 3) + ph2));
    const nu = Math.hypot(...ur) || 1, nv = Math.hypot(...vr) || 1;
    U.push(ur.map(x => x / nu)); V.push(vr.map(x => x / nv));
  }
  const noise = [];
  for (let i = 0; i < n; i++) { const row = []; for (let j = 0; j < p; j++) row.push(0.04 * (rng() * 2 - 1)); noise.push(row); }
  const M = [];
  for (let i = 0; i < n; i++) { const row = []; for (let j = 0; j < p; j++) { let v = noise[i][j]; for (let r = 0; r < R; r++) v += s[r] * U[r][i] * V[r][j]; row.push(v); } M.push(row); }
  const maxAbs = d3.max(M.flat(), v => Math.abs(v));
  const uMax = d3.max(U.flat(), v => Math.abs(v)) || 1;
  const vMax = d3.max(V.flat(), v => Math.abs(v)) || 1;
  const totalEnergy = s.reduce((a, b) => a + b * b, 0) + d3.sum(noise.flat(), v => v * v);
  return { n, p, R, s, U, V, M, maxAbs, uMax, vMax, totalEnergy };
}

recon = function (k) {
  const { n, p, U, V, s } = decomp;
  const out = [];
  for (let i = 0; i < n; i++) { const row = []; for (let j = 0; j < p; j++) { let v = 0; for (let r = 0; r < k; r++) v += s[r] * U[r][i] * V[r][j]; row.push(v); } out.push(row); }
  return out;
}

decompColor = d3.scaleSequential(d3.interpolateRgbBasis(["#0a7d63", "#f4f1ec", "#E78824"])).domain([-decomp.maxAbs, decomp.maxAbs])
factorColor = d3.scaleSequential(d3.interpolateRgbBasis(["#0a7d63", "#f4f1ec", "#E78824"])).domain([-1, 1])

drawDecomp = function (k) {
  const { n, p, M, U, V, s, uMax, vMax, totalEnergy } = decomp;
  const cw = 5, W = 1000, H = 240;
  const Mk = recon(k);
  const svg = d3.create("svg").attr("viewBox", `0 0 ${W} ${H}`)
    .style("width", "100%").style("height", "auto").style("font-family", "inherit");

  const block = (get, rows, cols, x, y, color) => {
    for (let i = 0; i < rows; i++) for (let j = 0; j < cols; j++)
      svg.append("rect").attr("x", x + j * cw).attr("y", y + i * cw)
        .attr("width", cw - 0.4).attr("height", cw - 0.4).attr("fill", color(get(i, j)));
  };
  const txt = (t, x, y, o = {}) => svg.append("text").attr("x", x).attr("y", y)
    .attr("font-size", o.size || 13).attr("font-weight", o.w || 700)
    .attr("fill", o.fill || "#333").attr("text-anchor", o.anchor || "start").text(t);

  const y0 = 44;
  txt("M  (n × p)", 10, y0 - 8);
  block((i, j) => M[i][j], n, p, 10, y0, decompColor);

  txt("≈", 10 + p * cw + 12, y0 + n * cw / 2 + 6, { size: 24, w: 800, anchor: "middle" });

  const xU = 10 + p * cw + 34;
  txt("U", xU, y0 - 8);
  block((i, r) => U[r][i] / uMax, n, k, xU, y0, factorColor);

  const xMul = xU + k * cw + 8;
  txt("×", xMul, y0 + n * cw / 2 + 5, { size: 18, w: 800, anchor: "middle" });

  const xV = xMul + 14;
  txt("V", xV, y0 - 8);
  block((r, j) => V[r][j] / vMax, k, p, xV, y0, factorColor);

  const xArr = xV + p * cw + 14;
  txt("→", xArr, y0 + n * cw / 2 + 6, { size: 22, w: 800, anchor: "middle" });

  const xR = xArr + 18;
  txt(`Rebuilt (k = ${k})`, xR, y0 - 8);
  block((i, j) => Mk[i][j], n, p, xR, y0, decompColor);

  const pct = (s.slice(0, k).reduce((a, b) => a + b * b, 0) / totalEnergy) * 100;
  const bx = 10, by = y0 + n * cw + 26, bw = 520, bh = 20;
  txt("Variation captured", bx, by - 6);
  svg.append("rect").attr("x", bx).attr("y", by).attr("width", bw).attr("height", bh).attr("rx", 5).attr("fill", "#eee");
  svg.append("rect").attr("x", bx).attr("y", by).attr("width", bw * pct / 100).attr("height", bh).attr("rx", 5).attr("fill", "#E78824");
  txt(`${pct.toFixed(0)}%`, bx + bw + 10, by + bh - 3, { w: 800, size: 16, fill: "#E78824" });

  const full = n * p, comp = k * (n + p);
  txt(`Stored: k·(n+p) = ${comp}  vs  n·p = ${full}   (${(comp / full * 100).toFixed(0)}% of the original — real p is in the thousands)`,
    bx, by + bh + 20, { size: 12, w: 600, fill: "#777" });

  return svg.node();
}

PCA — the main patterns

PCA looks at the data table on its own and asks: along which direction do the samples vary the most? That direction (PC₁) often reveals natural groupings — clans or subclasses — even though it was never told they exist.

panel("pca")

PCA is unsupervised. The biggest source of variation isn’t always what you care about — here the groups stay mixed, and the separation score is near zero.

PLS — patterns tied to an outcome

PLS uses the data table and an outcome (e.g. disease vs. healthy). It looks for directions that vary across samples and track that outcome so the groups we want are cleanly apart.

Same data, but the outcome guides the direction hence the separation score jumps up.

panel("pls")

Spin it yourself

Drag the angle, or hit a button to animate the projection line to what PCA vs. PLS would choose. Watch the separation score (top-right) rise and fall.

viewof theta = {
  const toDeg = v => ((Math.atan2(v.y, v.x) * 180 / Math.PI) % 180 + 180) % 180;
  const pcaA = toDeg(pcaDir);
  const plsA = toDeg(plsDir);

  const cont = html`<div style="display:flex;align-items:center;flex-wrap:wrap;gap:14px;font-size:18px;font-family:inherit;margin:4px 0 8px;"></div>`;
  const range = html`<input type="range" min="0" max="180" step="0.5" value="0" style="width:240px;accent-color:#E78824;">`;
  const val = html`<span style="font-weight:800;color:#E78824;min-width:2.8em;">0°</span>`;
  const bStyle = "padding:6px 14px;border:0;border-radius:8px;color:#fff;font-weight:700;cursor:pointer;font-family:inherit;font-size:15px;";
  const bPCA = html`<button style="${bStyle}background:#E78824;">▶ to PCA</button>`;
  const bPLS = html`<button style="${bStyle}background:#007E5A;">▶ to PLS</button>`;

  const set = v => {
    range.value = v;
    val.textContent = (+v).toFixed(0) + "°";
    cont.value = +v;
    cont.dispatchEvent(new Event("input", { bubbles: true }));
  };
  range.oninput = () => set(range.value);

  let raf;
  const animateTo = target => {
    cancelAnimationFrame(raf);
    const start = +cont.value, t0 = performance.now(), dur = 750;
    const step = now => {
      const k = Math.min(1, (now - t0) / dur);
      const e = k < 0.5 ? 2 * k * k : 1 - Math.pow(-2 * k + 2, 2) / 2;
      set(start + (target - start) * e);
      if (k < 1) raf = requestAnimationFrame(step);
    };
    raf = requestAnimationFrame(step);
  };
  bPCA.onclick = () => animateTo(pcaA);
  bPLS.onclick = () => animateTo(plsA);

  cont.value = 0;
  cont.append(
    html`<span style="font-weight:700;white-space:nowrap;">Projection angle:</span>`,
    range, val, bPCA, bPLS
  );
  return cont;
}

The project

  • Build a Julia package implementing PCA, PLS, PMD, and related decompositions.
  • Engineered for the very large matrices typical for metabolomics having thousands of metabolites across many samples.
  • A clean, consistent light package meant to be a shared foundation for further biostatistics packages in the department.
  • Why Julia: near-C performance with readable high-level.


The aim: a fast, reusable decomposition toolkit that other department projects can build on.
